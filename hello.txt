<h1>一、 通过urlconnection抓取信息：</h1>

步骤：<br /><span></span>1.获取url<br /><span></span>2.获取http请求<br /><span></span>3.获取状态码<br /><span></span>4.根据状态吗返回信息。<br /><br />

package com.soft.crawler;<br /><br /><br />
import java.io.BufferedReader;<br />
import java.io.InputStreamReader;<br />
import java.net.URL;<br />
import sun.net.www.protocol.http.HttpURLConnection;<br /><br /><br />
public class Crawler {<br /><span></span>public static void main(String[] args){<br /><span></span>String r;<br /><span></span>//1.新建url对象，表示要访问的网页<br /><span></span>try {<span> </span><br /><span></span>URL url = new URL("http://www.sina.com.cn");<br /><span></span>//2.建立http连接,返回连接对象urlconnection<br /><span></span>HttpURLConnection urlConnection = (HttpURLConnection) url.openConnection();<br /><span></span>//3.获取相应的http状态码，<br /><span></span>int responseCode= urlConnection.getResponseCode();<br /><span></span>//4.如果获取成功，从URLconnection对象获取输入流来获取请求网页的源代码<br /><span></span>if(responseCode == 200){<br /><span></span>BufferedReader reader = new BufferedReader(new InputStreamReader(urlConnection.getInputStream(), "utf-8"));<br /><span></span>while((r=reader.readLine())!=null){<br /><span></span>System.out.println(r);<br /><span></span>}<br /><span></span>}else{<br /><span></span>System.out.println("获取不到源代码 ，服务器响应代码为："+responseCode);<br /><span></span>}<br /><span></span>} catch (Exception e) {<br /><span></span>System.out.println("获取不到网页源码："+e);<br /><span></span>}<br />
}<br />
